디렉티브를 만들었다. 포커스를 잃을 때에만 바인딩을 하고 싶다.

function (UtilService) {
return {
  restrict: 'A',
  link: function (scope, iElem, iAttr) {
    iElem.unbind('input').unbind('keydown').unbind('change');
    iElem.bind('blur', function() {
      scope.$apply(iAttr.labelBlur);
    });
  }
};
}
