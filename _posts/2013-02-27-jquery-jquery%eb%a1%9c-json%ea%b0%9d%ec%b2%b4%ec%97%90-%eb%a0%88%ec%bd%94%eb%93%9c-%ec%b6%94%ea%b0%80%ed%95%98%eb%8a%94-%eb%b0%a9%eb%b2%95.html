아래와 같은 형식의 JSON데이터인 ListData에 새로운 레코드를 추가하려면 extend라는 메소드를 쓰면 된다.

jQuery.extend()

var ListData = {
  "main": {
    "label":"Main",
    "url":"#main"
  },
  "project": {
    "label":"Project",
    "url":"#project"
  },
  "settings": {
    "label":"Settings",
    "url":"#settings",
    "subnav":[
      {
        "label":"Privacy",
        "url":"#privacy"
      },
      {
        "label":"Security",
        "url":"#security"
      },
      {
        "label":"Advanced",
        "url":"#advanced"
      }
    ]
  }
};

$.extend(ListData, {"hello": { "label":"Hello", "url":"#hello" }});
