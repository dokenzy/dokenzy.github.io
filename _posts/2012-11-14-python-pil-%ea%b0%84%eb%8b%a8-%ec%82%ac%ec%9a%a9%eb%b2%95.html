인터페이스가 쉽고 이미 많은 튜토리얼이 있으니 간단한 사용법만 남긴다.

from PIL import Image
img = Image.open('s.png')
_xsize, _ysize = img.size
xsize = int(_xsize * 0.5)
ysize = int(_ysize * 0.5)

img2 = img.resize((xsize, ysize))#resize의 결과로 새로운 Image 객체를 반환한다.
img2.save('s2.png')#새로운 Image 객체인 img2를 저장한다.
