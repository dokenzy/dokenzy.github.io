간단한 자바스크립트를 이용하면 된다.

<!doctype html>
<html>
	<head>
		<script type="text/javascript">
			var svgSupport = (window.SVGSVGElement) ? true : false;
			if (svgSupport) alert("SVG Support");
			else alert("Oops");
		</script>
	</head>
	<body>
	</body>
</html>
